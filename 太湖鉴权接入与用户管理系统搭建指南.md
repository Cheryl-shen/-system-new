## 3. 实现鉴权中间件（自动执行）

根据检测到的框架类型，自动生成并写入鉴权中间件代码。

### 核心逻辑说明

太湖智能网关在用户登录后，会在转发给你的请求中注入以下 HTTP 请求头：

| 请求头 | 说明 |
|--------|------|
| `x-tai-identity` | 加密的用户身份信息（JWE 格式，用 APP_TOKEN 解密） |
| `timestamp` | 请求时间戳（秒） |
| `signature` | 网关签名（用于验证请求来源） |
| `x-rio-seq` | 请求序列号 |
| `staffname` | 员工英文名（仅兼容/明文模式，安全模式下为空） |
| `staffid` | 员工 ID（仅兼容/明文模式，安全模式下为空） |

**安全模式下**（推荐），需要解密 `x-tai-identity` 获取用户信息。解密后的 payload 包含：

```json
{
  "LoginName": "adamyide",       // 员工英文名
  "StaffId": 12345,              // 员工ID
  "ChineseName": "伊德日温",     // 中文名
  "Expiration": "2026-04-28T12:00:00+08:00"  // 过期时间
}
```

### 3.1 Node.js (Express/Koa) 实现

**创建文件** `middleware/auth.js`（路径根据项目结构自适应）：

```javascript
// middleware/auth.js
// 太湖鉴权中间件 - 解密 x-tai-identity 获取员工身份
const jose = require('jose');
const crypto = require('crypto');

// ============================================
// ⚠️ 配置项 - 需要替换为实际值
// ============================================
const TAI_APP_TOKEN = process.env.TAI_APP_TOKEN || '请替换为太湖应用Token';
const KEY_BYTES = Buffer.from(TAI_APP_TOKEN);
const IDENTITY_SAFE_MODE = true; // 安全模式，推荐开启

// ============================================
// 签名校验
// ============================================
function checkSignature(key, timestampSeconds, signature, extHeaders) {
  if (!timestampSeconds || isNaN(timestampSeconds)) {
    return false;
  }
  // 校验时间偏差不超过 3 分钟
  if (Math.abs(parseInt(timestampSeconds, 10) * 1000 - Date.now()) > 180000) {
    return false;
  }
  const hash = crypto.createHash('sha256');
  hash.update(timestampSeconds + key + extHeaders.join(',') + timestampSeconds);
  return signature.toLowerCase() === hash.digest('hex').toLowerCase();
}

// ============================================
// 解密身份
// ============================================
async function decodeIdentity(encryptedHeader, keyBytes) {
  const dec = await jose.compactDecrypt(encryptedHeader, keyBytes);
  const payload = JSON.parse(dec.plaintext);

  // 校验 token 过期时间（3分钟缓冲）
  const exp = new Date(payload.Expiration);
  if (new Date() - 3 * 60 * 1000 > exp) {
    throw new Error('Identity token expired');
  }
  return payload;
}

// ============================================
// Express 中间件
// ============================================
async function taiAuthMiddleware(req, res, next) {
  try {
    const headers = req.headers;

    // 构造签名校验参数
    let extHeaders;
    if (IDENTITY_SAFE_MODE) {
      extHeaders = [headers['x-rio-seq'] || '', '', '', ''];
    } else {
      extHeaders = [
        headers['x-rio-seq'] || '',
        headers['staffid'] || '',
        headers['staffname'] || '',
        headers['x-ext-data'] || ''
      ];
    }

    // 1. 校验网关签名
    if (!checkSignature(TAI_APP_TOKEN, headers['timestamp'], headers['signature'], extHeaders)) {
      return res.status(403).json({ error: '网关签名校验失败，请通过内网访问' });
    }

    // 2. 解密用户身份
    const taiIdentity = headers['x-tai-identity'];
    if (taiIdentity) {
      const payload = await decodeIdentity(taiIdentity, KEY_BYTES);
      req.user = {
        staffId: payload.StaffId,
        loginName: payload.LoginName,
        chineseName: payload.ChineseName || '',
        expiration: payload.Expiration
      };
    } else if (!IDENTITY_SAFE_MODE && headers['staffname']) {
      // 兼容模式回退
      req.user = {
        staffId: parseInt(headers['staffid'], 10) || 0,
        loginName: headers['staffname'],
        chineseName: '',
        expiration: null
      };
    } else {
      return res.status(401).json({ error: '未检测到用户身份信息' });
    }

    next();
  } catch (error) {
    console.error('[TAI Auth] 鉴权失败:', error.message);
    return res.status(401).json({ error: '鉴权失败: ' + error.message });
  }
}

// ============================================
// 免登录路径（健康检查等不需要鉴权的路径）
// ============================================
const PUBLIC_PATHS = [
  '/health',
  '/ping',
  '/favicon.ico'
];

function taiAuthWithPublicPaths(req, res, next) {
  if (PUBLIC_PATHS.some(p => req.path.startsWith(p))) {
    return next();
  }
  return taiAuthMiddleware(req, res, next);
}

module.exports = { taiAuthMiddleware, taiAuthWithPublicPaths };
```

### 3.2 Python (Flask/Django) 实现

**创建文件** `middleware/tai_auth.py`：

```python
# middleware/tai_auth.py
# 太湖鉴权中间件 - Python 版本
import os
import json
import time
import hashlib
from functools import wraps
from jwcrypto import jwe, jwk

TAI_APP_TOKEN = os.environ.get('TAI_APP_TOKEN', '请替换为太湖应用Token')
IDENTITY_SAFE_MODE = True

def check_signature(key, timestamp_seconds, signature, ext_headers):
    """校验太湖网关签名"""
    if not timestamp_seconds or not timestamp_seconds.isdigit():
        return False
    if abs(int(timestamp_seconds) * 1000 - int(time.time() * 1000)) > 180000:
        return False
    data = timestamp_seconds + key + ','.join(ext_headers) + timestamp_seconds
    calculated = hashlib.sha256(data.encode()).hexdigest()
    return signature.lower() == calculated.lower()

def decode_identity(encrypted_header, token):
    """解密 x-tai-identity 获取员工信息"""
    key = jwk.JWK(kty='oct', k=jwk.base64url_encode(token.encode()))
    jwe_obj = jwe.JWE()
    jwe_obj.deserialize(encrypted_header, key)
    payload = json.loads(jwe_obj.payload)
    return payload

# Flask 装饰器版本
def require_tai_auth(f):
    """Flask 路由鉴权装饰器"""
    @wraps(f)
    def decorated(*args, **kwargs):
        from flask import request, jsonify, g

        headers = request.headers
        ext_headers = [headers.get('X-Rio-Seq', ''), '', '', ''] if IDENTITY_SAFE_MODE else [
            headers.get('X-Rio-Seq', ''),
            headers.get('Staffid', ''),
            headers.get('Staffname', ''),
            headers.get('X-Ext-Data', '')
        ]

        if not check_signature(TAI_APP_TOKEN, headers.get('Timestamp', ''),
                                headers.get('Signature', ''), ext_headers):
            return jsonify({'error': '网关签名校验失败'}), 403

        tai_identity = headers.get('X-Tai-Identity')
        if tai_identity:
            try:
                payload = decode_identity(tai_identity, TAI_APP_TOKEN)
                g.user = {
                    'staff_id': payload.get('StaffId'),
                    'login_name': payload.get('LoginName'),
                    'chinese_name': payload.get('ChineseName', ''),
                }
            except Exception as e:
                return jsonify({'error': f'鉴权失败: {str(e)}'}), 401
        else:
            return jsonify({'error': '未检测到用户身份'}), 401

        return f(*args, **kwargs)
    return decorated
```

### 3.3 Go (Gin/Echo) 实现

**创建文件** `middleware/tai_auth.go`：

```go
// middleware/tai_auth.go
package middleware

import (
	"crypto/sha256"
	"encoding/json"
	"fmt"
	"math"
	"net/http"
	"os"
	"strconv"
	"strings"
	"time"

	"github.com/gin-gonic/gin"
	"github.com/lestrrat-go/jwx/v2/jwe"
	"github.com/lestrrat-go/jwx/v2/jwa"
)

var (
	taiAppToken      = os.Getenv("TAI_APP_TOKEN")
	identitySafeMode = true
)

type TaiIdentity struct {
	StaffId     int    `json:"StaffId"`
	LoginName   string `json:"LoginName"`
	ChineseName string `json:"ChineseName"`
	Expiration  string `json:"Expiration"`
}

func checkSignature(key, timestamp, signature string, extHeaders []string) bool {
	if timestamp == "" {
		return false
	}
	ts, err := strconv.ParseInt(timestamp, 10, 64)
	if err != nil {
		return false
	}
	if math.Abs(float64(ts*1000-time.Now().UnixMilli())) > 180000 {
		return false
	}
	data := timestamp + key + strings.Join(extHeaders, ",") + timestamp
	hash := sha256.Sum256([]byte(data))
	return strings.EqualFold(fmt.Sprintf("%x", hash), signature)
}

func TaiAuthMiddleware() gin.HandlerFunc {
	return func(c *gin.Context) {
		var extHeaders []string
		if identitySafeMode {
			extHeaders = []string{c.GetHeader("X-Rio-Seq"), "", "", ""}
		} else {
			extHeaders = []string{
				c.GetHeader("X-Rio-Seq"),
				c.GetHeader("Staffid"),
				c.GetHeader("Staffname"),
				c.GetHeader("X-Ext-Data"),
			}
		}

		if !checkSignature(taiAppToken, c.GetHeader("Timestamp"),
			c.GetHeader("Signature"), extHeaders) {
			c.AbortWithStatusJSON(http.StatusForbidden,
				gin.H{"error": "网关签名校验失败"})
			return
		}

		taiIdentity := c.GetHeader("X-Tai-Identity")
		if taiIdentity == "" {
			c.AbortWithStatusJSON(http.StatusUnauthorized,
				gin.H{"error": "未检测到用户身份"})
			return
		}

		decrypted, err := jwe.Decrypt([]byte(taiIdentity),
			jwe.WithKey(jwa.A256GCM, []byte(taiAppToken)))
		if err != nil {
			c.AbortWithStatusJSON(http.StatusUnauthorized,
				gin.H{"error": "身份解密失败"})
			return
		}

		var identity TaiIdentity
		if err := json.Unmarshal(decrypted, &identity); err != nil {
			c.AbortWithStatusJSON(http.StatusUnauthorized,
				gin.H{"error": "身份信息解析失败"})
			return
		}

		c.Set("user", identity)
		c.Next()
	}
}
```

### 5.2 用户自动注册与登录确认中间件

> **核心需求**：每次登录需要手动确认（而不是静默自动登录）。

在步骤 3 的鉴权中间件之后，添加一层**登录确认中间件**：

#### Node.js 实现

**创建文件** `middleware/loginConfirm.js`：

```javascript
// middleware/loginConfirm.js
// 登录确认中间件 - 要求用户手动确认每次登录
const db = require('../db'); // 根据实际项目调整数据库引用

// Session 存储（可替换为 Redis）
const activeSessions = new Map();

// 会话有效期（毫秒）
const SESSION_TTL = 8 * 60 * 60 * 1000; // 8小时

function loginConfirmMiddleware(req, res, next) {
  // 免确认路径
  const skipPaths = ['/health', '/ping', '/favicon.ico', '/api/auth/confirm', '/auth/login'];
  if (skipPaths.some(p => req.path.startsWith(p))) {
    return next();
  }

  const user = req.user; // 由太湖鉴权中间件注入
  if (!user) {
    return res.status(401).json({ error: '未登录' });
  }

  // 检查是否有已确认的有效会话
  const sessionKey = `${user.loginName}_session`;
  const session = activeSessions.get(sessionKey);

  if (session && (Date.now() - session.confirmedAt) < SESSION_TTL) {
    // 会话有效，放行
    req.dbUser = session.dbUser;
    return next();
  }

  // 会话不存在或已过期 → 需要用户手动确认登录
  // 返回特定状态码，前端据此显示登录确认页面
  return res.status(449).json({
    error: 'LOGIN_CONFIRMATION_REQUIRED',
    message: '请确认登录',
    user: {
      loginName: user.loginName,
      chineseName: user.chineseName,
      staffId: user.staffId
    }
  });
}

// 登录确认接口
async function confirmLogin(req, res) {
  const user = req.user;
  if (!user) {
    return res.status(401).json({ error: '未登录' });
  }

  try {
    // 1. 查找或创建用户记录
    let dbUser = await db.findUserByLoginName(user.loginName);

    if (!dbUser) {
      // 首次登录，自动创建用户
      dbUser = await db.createUser({
        staffId: user.staffId,
        loginName: user.loginName,
        chineseName: user.chineseName,
        role: 'user',
        status: 'active'
      });
    }

    // 2. 检查用户是否被禁用
    if (dbUser.status === 'disabled') {
      return res.status(403).json({
        error: '您的账号已被管理员禁用，请联系管理员'
      });
    }

    // 3. 更新最近登录时间
    await db.updateLastLogin(dbUser.id);

    // 4. 记录登录日志
    await db.createLoginLog({
      userId: dbUser.id,
      loginName: user.loginName,
      ipAddress: req.headers['x-forwarded-for'] || req.ip,
      userAgent: req.headers['user-agent'] || ''
    });

    // 5. 创建已确认会话
    const sessionKey = `${user.loginName}_session`;
    activeSessions.set(sessionKey, {
      dbUser: dbUser,
      confirmedAt: Date.now()
    });

    return res.json({
      success: true,
      message: '登录确认成功',
      user: {
        id: dbUser.id,
        loginName: dbUser.loginName,
        chineseName: dbUser.chineseName,
        role: dbUser.role,
        status: dbUser.status
      }
    });
  } catch (error) {
    console.error('[Login Confirm] 错误:', error);
    return res.status(500).json({ error: '登录处理失败' });
  }
}

// 登出接口
function logout(req, res) {
  const user = req.user;
  if (user) {
    activeSessions.delete(`${user.loginName}_session`);
  }
  return res.json({ success: true, message: '已登出' });
}

module.exports = { loginConfirmMiddleware, confirmLogin, logout };
```

## 附录 C：常见问题

### Q1：用户每次打开网页都要确认登录吗？
不是。会话有效期为 8 小时（可配置 `SESSION_TTL`）。在会话有效期内，用户无需重复确认。关闭浏览器或会话过期后，下次访问需要重新确认。

### Q2：如何修改会话有效期？
在 `middleware/loginConfirm.js` 中修改 `SESSION_TTL` 常量。

### Q3：站点原本已有鉴权，会冲突吗？
建议在太湖平台配置站点时，取消勾选"统一登录页员工鉴权"，使用本文档的方案代替。如果要保留原有鉴权，需要协调两套登录逻辑。

### Q4：线上支持联系方式？
太湖平台问题联系 TEG 技术支持：企业微信搜索 `2000(TEG技术支持)`。
